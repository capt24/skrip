<?php
$folder = 'galery/';  // Pastikan folder ini ada dan berisi gambar
$cacheFolder = 'cache/';
$cacheFile = $cacheFolder . 'gallery_cache.json';

if (!is_dir($cacheFolder)) {
    mkdir($cacheFolder, 0777, true); // Buat folder cache jika belum ada
}

$allowedExtensions = ['jpg', 'jpeg', 'png', 'bmp', 'gif', 'webp'];

if (!file_exists($cacheFile) || time() - filemtime($cacheFile) > 86400) {
    $files = array_diff(scandir($folder), ['..', '.']);
    $files = array_filter($files, function ($file) use ($allowedExtensions, $folder) {
        return in_array(strtolower(pathinfo($file, PATHINFO_EXTENSION)), $allowedExtensions);
    });
    file_put_contents($cacheFile, json_encode(array_values($files))); // Simpan daftar gambar ke cache
} else {
    $files = json_decode(file_get_contents($cacheFile), true);
}

                index = (index + step + images.length) % images.length;

            function openPopup(imgElement) {
                const popup = document.getElementById('popup');
                const popupImg = document.getElementById('popup-img');
                popupImg.src = imgElement.getAttribute('data-full');
                popup.style.display = 'flex';
