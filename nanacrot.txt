<script>
	$('#form-login').submit(function(e) {
		e.preventDefault();
		$.ajax({
			type: 'POST',
			url: 'ceklogin.php?pg=login',
			data: $(this).serialize(),

			success: function(data) {
				var json = $.parseJSON(data);
				if (json.pesan == 'ok') {
					iziToast.success({
						title: 'Gagah Men !!!',
						message: 'Ngana So Berhasil Login',
						position: 'topRight'
					});
					setTimeout(function() {
						window.location.href = "user";
					}, 2000);

				} else {
					iziToast.error({
						title: 'Maaf !!!',
						message: json.pesan,
						position: 'topCenter'
					});
				}

			}
		});
		return false;
	});
